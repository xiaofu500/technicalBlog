tar 压缩包

  • tar 是 "Tape Archive" 的缩写，它是一个在 Unix-like 操作系统（包括 Linux 和 macOS）中非常常用的文件归档工具。 它的主要功能是将多个文件和目录打包成一个单一的文件，这个单一的文件就叫做 "tarball" 或 "tar 归档文件"。

 * 上传文件
 * @param destnation 上传文件的存储目录

      /** 存储配置
       *  destination: 上传文件的存储目录
       * 上传文件存储配置
       * @param req 请求对象
       * @param file 上传的文件对象
       * @param cb 回调函数，用于指定上传文件的存储路径和文件名
       */

        destination: './upload',
        filename: (req, file, cb) => {
          console.log(file, req);
          const fileName = `${Date.now()}-${file.originalname}`;
          cb(null, fileName);
        },

文件下载 ​

typescript
/**
 * 文件下载
 * @param fileName 文件名
 * @param res 响应对象
 */
import { Response } from 'express';

/**
 * 文件下载
 * @param fileName 文件名
 * @param res 响应对象
 */
download(filename: string) {
    const uploadDir = this.configService.get<string>('file.uploadPath');
    if (!uploadDir) {
      throw new Error('uploadDir not found');
    }
    const filePath = path.join(uploadDir, filename);
    if (!fs.existsSync(filePath)) {
      throw new Error('file not found');
    }

    if (!filePath) {
      throw new Error('tarStream not found');
    }
    return {
      filename,
      fsStream: fs.createReadStream(filePath),
    };
  }
