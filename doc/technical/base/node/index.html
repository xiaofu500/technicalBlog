node 基础知识 ​

package.json 配置 ​

配置文件内容

typescript
name:"@xiaofu/core", //xiaofu 为团队名称,core 为项目名称
version:"1.0.0",
main:"index.js",// 入口文件,一般设置为打包后的index.js
module :"index.js",// 模块文件,一般设置为打包后的index.js
description:"",// 项目描述
keywords:[""],// 项目关键词
private: true,// 是否为私有项目
scripts:{
  "start":"node index.js" //项目命令文件
}

{
  "engines": {// 项目运行环境
    "runtime": { // 项目运行时环境
      "name": "node",
      "version": "^24.11.0",
      "onFail": "download" // 项目运行时环境失败时的处理方式,download表示下载安装
    }
  }
}
二、项目依赖配置
"dependencies": { // 项目依赖，生产环境和开发环境都需要的依赖
    "button": "workspace:1.0.0"
  }
  "devDependencies": { // 项目依赖，开发环境需要的依赖,不会打包到dist 文件中
    "typescript": "^5.5.4"
  }

  "optionalDependencies": { // 项目依赖，可选依赖,如果安装失败,不会影响项目运行
    "button": "workspace:1.0.0"
  }
  "peerDependencies": { // 项目依赖，同伴依赖,如果安装失败,会影响项目运行
    "button": "workspace:1.0.0"
  }
  "bundleDependencies": { // 项目依赖，打包依赖,会打包到dist 文件中
    "button": "workspace:1.0.0" //应用场景: 项目依赖的模块,在项目运行时,需要被打包到dist 文件中
  }
三、项目发布配置
   "publishConfig": {
        "main": "lib/index.js",
        "typings": "lib/index.d.ts"
    }
    或者
    "publishConfig": {
        "directory": "dist" //完整的被打包文件，包含package.json文件
       "linkDirectory": true // 当设置为 true 时，项目将在本地开发期间从 publishConfig.directory 位置进行符号链接。
    }
