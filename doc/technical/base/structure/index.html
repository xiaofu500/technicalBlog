2.链表 ​

适用与需要频繁添加和删除操作的场景

typescript
class ListNode<T> {
  value: T;
  next: ListNode<T> | null = null;
  constructor(value: T) {
    this.value = value;
  }
}
export class LinkedList<T> {
  private head: ListNode<T> | null = null;
  constructor() {
    this.head = null;
  }
  append(value: T) {
    const newNode = new ListNode(value);
    if (!this.head) {
      this.head = newNode;
    } else {
      let current = this.head;
      while (current.next) {
        current = current.next;
      }
      current.next = newNode;
    }
    return this.head;
  }
  toArray(): T[] {
    const result: T[] = [];
    let current = this.head;
    while (current) {
      result.push(current.value);
      current = current.next;
    }
    return result;
  }
}

4.树 ​

理解：树是一种非线性数据结构，由节点组成，每个节点可以有多个子节点。树的每个节点都有一个父节点，除了根节点没有父节点。树的每个节点都可以看作是一个子树的根节点。

理解插入 6 的过程

1. 初始化
新节点：6
队列初始化为根节点：[1]

1. 初始化
新节点：6
队列初始化为根节点：[1]

2. 第一次循环
队列：[1]
取出队头节点：1
检查1的左子节点：有（2），入队
检查1的右子节点：有（3），入队
队列变为：[2, 3]
1的左右都满，继续下一轮

3. 第二次循环
队列：[2, 3]
取出队头节点：2
检查2的左子节点：有（4），入队
检查2的右子节点：有（5），入队
队列变为：[3, 4, 5]
2的左右都满，继续下一轮

4. 第三次循环
队列：[3, 4, 5]

检查3的左子节点：没有
将新节点6插入到3的左子节点

class TreeNode<T> {
  value: T;
  children: TreeNode<T>[] = [];
  constructor(value: T) {
    this.value = value;
  }
}
export class Tree<T> {
  root: TreeNode<T> | null = null;
  constructor() {
    this.root = null;
  }
  insert(value: T) {
    if (!this.root) {
      this.root = new TreeNode(value);
      return;
    }
    // 层序遍历，找到第一个没有子节点的节点，添加到children中
    const queue = [this.root];
    /**
     *
     * 分两种情况，当前节点是否有子节点：
     * 1。没有子节点，创建节点添加到children中
     * 2.有子节点，添加到第一个子节点的children中
     */
    while (queue.length) {
      const current = queue.shift()!;
      if (current.children.length < 2) {
        current.children.push(new TreeNode(value));
        return;
      }
      queue.push(...current.children);
    }
  }
}

5.堆（小根堆-react 渲染调度） ​

堆为树的变种，最小堆，就是每个节点的值都小于等于其子节点的值。

  • 规律记忆： 1.父节点的下标为 (i - 1) / 2 2.左子节点的下标为 2i + 1 3.右子节点的下标为 2i + 2

  • 应用场景：

    1. 优先队列：根据元素的优先级来确定处理顺序，例如任务调度、图算法中的 Dijkstra 算法等。
    2. 排序：通过构建最小堆或最大堆，可以实现高效的排序算法，如堆排序。
    3. 图算法：在图算法中，最小堆常用于实现 Dijkstra 算法，找到从起始节点到其他节点的最短路径。
typescript
export class MinHeap<T> {
  private heap: T[] = [];
  constructor() {
    this.heap = [];
  }
  insert(value: T) {
    this.heap.push(value);
    this.bubbleUp();
  }
  // 从堆中提取最小值
  extractMin(): T | undefined {
    if (this.heap.length === 0) {
      return undefined;
    }
    if (this.heap.length === 1) {
      return this.heap.pop();
    }
    //堆化处理
    const min = this.heap[0];
    this.heap[0] = this.heap.pop()!;
    this.bubbleDown();
    return min;
  }
  bubbleUp() {
    let currentIndex = this.heap.length - 1;
    const value = this.heap[currentIndex];
    while (currentIndex > 0) {
      const parentIndex = Math.floor((currentIndex - 1) / 2);
      //子节点的值大于等于父节点的值，不需要交换
      const parentValue = this.heap[parentIndex];

      if (value >= parentValue) {
        break;
      }
      this.heap[currentIndex] = parentValue;
      currentIndex = parentIndex;
    }
    this.heap[currentIndex] = value;

}
bubbleDown() {
let currentIndex = 0;
const value = this.heap[currentIndex];
while (currentIndex < this.heap.length) {
const leftChildIndex = 2 _ currentIndex + 1;
const rightChildIndex = 2 _ currentIndex + 2;
let minChildIndex = currentIndex;
if (leftChildIndex < this.heap.length && this.heap[leftChildIndex] < this.heap[minChildIndex]) {
minChildIndex = leftChildIndex;
}
if (rightChildIndex < this.heap.length && this.heap[rightChildIndex] < this.heap[minChildIndex]) {
minChildIndex = rightChildIndex;
}
if (minChildIndex === currentIndex) {
break;
}
this.heap[currentIndex] = this.heap[minChildIndex];
currentIndex = minChildIndex;
}
this.heap[currentIndex] = value;
}
}
