2.链表 ​

适用与需要频繁添加和删除操作的场景

typescript
class ListNode<T> {
  value: T;
  next: ListNode<T> | null = null;
  constructor(value: T) {
    this.value = value;
  }
}
export class LinkedList<T> {
  private head: ListNode<T> | null = null;
  constructor() {
    this.head = null;
  }
  append(value: T) {
    const newNode = new ListNode(value);
    if (!this.head) {
      this.head = newNode;
    } else {
      let current = this.head;
      while (current.next) {
        current = current.next;
      }
      current.next = newNode;
    }
    return this.head;
  }
  toArray(): T[] {
    const result: T[] = [];
    let current = this.head;
    while (current) {
      result.push(current.value);
      current = current.next;
    }
    return result;
  }
}

3.树 ​

理解：树是一种非线性数据结构，由节点组成，每个节点可以有多个子节点。树的每个节点都有一个父节点，除了根节点没有父节点。树的每个节点都可以看作是一个子树的根节点。

理解插入 6 的过程

1. 初始化
新节点：6
队列初始化为根节点：[1]

1. 初始化
新节点：6
队列初始化为根节点：[1]

2. 第一次循环
队列：[1]
取出队头节点：1
检查1的左子节点：有（2），入队
检查1的右子节点：有（3），入队
队列变为：[2, 3]
1的左右都满，继续下一轮

3. 第二次循环
队列：[2, 3]
取出队头节点：2
检查2的左子节点：有（4），入队
检查2的右子节点：有（5），入队
队列变为：[3, 4, 5]
2的左右都满，继续下一轮

4. 第三次循环
队列：[3, 4, 5]
取出队头节点：3
检查3的左子节点：没有
将新节点6插入到3的左子节点
插入完成，退出循环

typescript
class TreeNode<T> {
  value: T;
  children: TreeNode<T>[] = [];
  constructor(value: T) {
    this.value = value;
  }
}
export class Tree<T> {
  root: TreeNode<T> | null = null;
  constructor() {
    this.root = null;
  }
  insert(value: T) {
    if (!this.root) {
      this.root = new TreeNode(value);
      return;
    }
    // 层序遍历，找到第一个没有子节点的节点，添加到children中
    const queue = [this.root];
    /**
     *
     * 分两种情况，当前节点是否有子节点：
     * 1。没有子节点，创建节点添加到children中
     * 2.有子节点，添加到第一个子节点的children中
     */
    while (queue.length) {
      const current = queue.shift()!;
      if (current.children.length < 2) {
        current.children.push(new TreeNode(value));
        return;
      }
      queue.push(...current.children);
    }
  }
}

总结： 在思维推导过程中，忽略队列中数据的出入状况，从而导致条件判断错误。 插值在于搞懂队列的推演过程

4.堆 ​

堆为树的变种，最小堆，就是每个节点的值都小于等于其子节点的值。

typescript
class MinHeap<T> {
  heap: T[] = [];
  constructor() {
    this.heap = [];
  }
  private parent(index: number) {
    return Math.floor((index - 1) / 2);
  }
  private leftChild(index: number) {
    return index * 2 + 1;
  }
  private rightChild(index: number) {
    return index * 2 + 2;
  }
  bubbleUp(index: number) {
    while (index > 0) {
      const parentIndex = this.parent(index);
      if (this.heap[index] < this.heap[parentIndex]) {
        [this.heap[index], this.heap[parentIndex]] = [this.heap[parentIndex], this.heap[index]];
        index = parentIndex;
      } else {
        break;
      }
    }
  }
  insert(value: T) {
    this.heap.push(value);
    let currentIndex = this.heap.length - 1;
    this.bubbleUp(currentIndex);
  }
  bubbleDown(index: number) {
    while (this.leftChild(index) < this.heap.length) {
      const leftChildIndex = this.leftChild(index);
      const rightChildIndex = this.rightChild(index);
      let smallerChildIndex = leftChildIndex;
      if (rightChildIndex < this.heap.length && this.heap[rightChildIndex] < this.heap[leftChildIndex]) {
        smallerChildIndex = rightChildIndex;
      }
      if (this.heap[index] < this.heap[smallerChildIndex]) {
        break;
      }
      [this.heap[index], this.heap[smallerChildIndex]] = [this.heap[smallerChildIndex], this.heap[index]];
      index = smallerChildIndex;
    }
  }
  getHeap() {
    return this.heap;
  }
  findMax() {
    return this.heap[0];
  }
  sort(array: T[]) {
    array.forEach((item) => {
      this.insert(item);
    });
    const sortedArray: T[] = [];
    while (this.heap.length) {
      sortedArray.push(this.heap[0]);
      this.heap[0] = this.heap[this.heap.length - 1];
      this.heap.pop();
      this.bubbleDown(0);
    }
    return sortedArray;
  }
  getHeap() {
    return this.heap;
  }
}
